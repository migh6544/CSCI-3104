\section{Standard 16- Analyzing Code III: Writing Recurrences}
\begin{required}
Write down a recurrence for the runtime complexity of this algorithm. Clearly justify your answer. You are \textbf{not} being asked to solve the recurrence.

\begin{algorithm}
\caption{Recurrences}\label{alg:NestedIndependent}
\begin{algorithmic}[1]
\Procedure{Foo1}{$\text{Integer } n$}
\If{$n < 5$}
\Return
\EndIf

\State $\text{Foo1}(n/7)$
\State $\text{Foo1}(n/7)$
\State $\text{Foo1}(n/7)$
\State $\text{Foo1}(n/7)$
\State $\text{Foo1}(n/7)$ \\

\For{$i \gets 1; i \leq n; i \gets i * 8$}
	\State $\textbf{print } (2 * i)$
\EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}
\end{required}

\begin{proof}[Answer:] \
\begin{itemize}
\item \textbf{We hit the base case when $(n < 5)$ using only a constant number of atomic operations $\to \Theta(1)$}
\item \textbf{When $(n \geq 5)$ there are ($5$) recursive calls, each of which with input ($\frac{n}{7}$), with term $\to 5T(\frac{n}{7})$}
\item \textbf{The non-recursive part of the function has an upper-bound of $n$ and multiplies the index $i$ by $8$ each time, so the total non-recursive work is $\Theta(\log_8n)$ $\to \Theta(\log n)$}
\end{itemize}
\item \textbf{Thus the runtime complexity function is:}
\begin{align*}
T(n) &= \begin{cases}
\Theta(1) & : n < 5, \\
5T(\frac{n}{7}) + \Theta(\log(n)) & : n \geq 5.
\end{cases}
\end{align*}
%Your answer here
\end{proof}
